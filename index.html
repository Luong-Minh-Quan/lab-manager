// Kết nối với Supabase (Sử dụng API Key bạn lấy từ cài đặt Supabase) const supabase = supabase.createClient('URL_CUA_BAN', 'ANON_KEY_CUA_BAN') // Hàm Đăng ký async function signUp(email, password) { const { data, error } = await supabase.auth.signUp({ email, password }) if (error) alert("Lỗi: " + error.message) else alert("Kiểm tra email để xác nhận tài khoản!") } // Hàm Đăng ký sử dụng phòng Lab async function bookLab(labName, date) { const { data, error } = await supabase .from('bookings') .insert([{ lab_name: labName, booking_date: date }]) if (error) alert("Không thể đăng ký!") else alert("Đăng ký phòng Lab thành công!") }
